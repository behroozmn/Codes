        tkinter.messagebox.showerror('Error', 'something went wrong')
    elif ms == 'division zero error':
        tkinter.messagebox.showerror('Division Error', 'Can Not Divide By 0')

        value = float(num1.get()) + float(num2.get())

    try:
        value = float(num1.get()) - float(num2.get())

    except:
